from external_tools.model_conversion.converters import FormatConverter

# Initialize
manager = ModelManager(cache_dir="external_tools/surya/models_cache")

# Batch OCR
job = manager.batch_ocr(
    input_dir="F:\images",
    output_dir="output",
    engine="surya",
    languages=["he", "ar"]
)

# Monitor progress
while job.status != "completed":
    print(f"Progress: {job.progress}%")
    time.sleep(1)

# Convert model
job = manager.convert_model(
    source_engine="surya",
    source_format="pytorch",
    target_engine="kraken",
    target_format="onnx"
)

# Convert results
FormatConverter.pagexml_to_alto(
    pagexml_path="result.pagexml",
    output_path="result.alto.xml"
)
